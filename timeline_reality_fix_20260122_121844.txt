// REAL PROGRESS CALCULATION - Copy confidence meter approach
const calculateRealTimelineProgress = async (userId: string, userProfile?: any) => {
  try {
    const supabase = createClient();
    
    const [
      { data: messages },
      { data: documents },
      { data: progress },
      { data: expertSessions }
    ] = await Promise.all([
      supabase.from('messages').select('count').eq('user_id', userId).single(),
      supabase.from('user_documents').select('status').eq('user_id', userId),
      supabase.from('user_progress').select('*').eq('user_id', userId).single(),
      supabase.from('expert_sessions').select('count').eq('user_id', userId).single()
    ]);

    // Calculate REAL progress percentage (0-100)
    let realProgress = 0;
    
    // 1. Profile Completeness (25% max)
    if (userProfile?.country) realProgress += 8;
    if (userProfile?.destination_country) realProgress += 8;
    if (userProfile?.visa_type) realProgress += 9;

    // 2. Document Readiness (35% max)
    const approvedDocs = documents?.filter(d => d.status === 'approved').length || 0;
    const totalDocs = 8; // Target
    realProgress += (approvedDocs / totalDocs) * 35;

    // 3. AI Engagement (20% max)
    const messageCount = messages?.count || 0;
    if (messageCount >= 10) realProgress += 20;
    else if (messageCount >= 5) realProgress += 15;
    else if (messageCount >= 1) realProgress += 8;

    // 4. Expert Support (10% max)
    const expertCount = expertSessions?.count || 0;
    if (expertCount >= 3) realProgress += 10;
    else if (expertCount >= 1) realProgress += 5;

    // 5. Existing Progress (10% max)
    const existingProgress = progress?.progress_percentage || 0;
    realProgress += Math.min(existingProgress * 0.1, 10);

    return Math.min(Math.round(realProgress), 100);
  } catch (error) {
    console.error('Error calculating real timeline progress:', error);
    return 0; // Conservative fallback
  }
};

// DYNAMIC STAGE THRESHOLDS - Based on real user actions
const getDynamicStageThresholds = (realProgress: number, profile: any, documents: any[], messages: any[]) => {
  const approvedDocs = documents?.filter(d => d.status === 'approved').length || 0;
  const messageCount = messages?.count || 0;

  return {
    // Stage 1: Getting Started (0-25%)
    onboarding: {
      completed: realProgress >= 25,
      inProgress: realProgress >= 10,
      // Completed when: Profile complete + some engagement
      isCompleted: !!(profile?.country && profile?.destination_country && profile?.visa_type && messageCount >= 1)
    },
    
    // Stage 2: Document Preparation (25-50%)
    documentPrep: {
      completed: realProgress >= 50,
      inProgress: realProgress >= 30,
      // Completed when: 4+ docs approved + active engagement
      isCompleted: approvedDocs >= 4 && messageCount >= 3
    },
    
    // Stage 3: Application Submission (50-75%)
    submission: {
      completed: realProgress >= 75,
      inProgress: realProgress >= 60,
      // Completed when: 6+ docs + expert consultation
      isCompleted: approvedDocs >= 6 && messageCount >= 5
    },
    
    // Stage 4: Interview Prep (75-90%)
    interview: {
      completed: realProgress >= 90,
      inProgress: realProgress >= 80,
      // Completed when: Almost ready + expert sessions
      isCompleted: realProgress >= 88
    },
    
    // Stage 5: Final Prep (90-100%)
    finalPrep: {
      completed: realProgress >= 100,
      inProgress: realProgress >= 95,
      isCompleted: realProgress >= 98
    },
    
    // Stage 6: Approval (100%)
    approval: {
      completed: realProgress >= 100,
      isCompleted: realProgress >= 100
    }
  };
};
